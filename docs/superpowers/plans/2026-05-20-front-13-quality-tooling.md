# Front 13 Quality Tooling Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add reliable quality tooling for SeCuida: explicit script names, ESLint, Prettier, content validation, broader tests, and one `pnpm run check` merge gate.

**Architecture:** Keep tooling at the repository boundary. Runtime app code remains independent of validation scripts, while `scripts/validate-flows.ts` imports existing content/domain modules to validate the same flows and resources the app uses.

**Tech Stack:** React 19, Vite 6, TypeScript 5.8, pnpm, Vitest, Testing Library, ESLint flat config, Prettier, tsx.

- [ ] **Step 5: Create `scripts/validate-flows.ts`**

```ts
import { existsSync, readdirSync, readFileSync } from 'node:fs';
import path from 'node:path';
import { fileURLToPath } from 'node:url';
import { restRecoveryFlow } from '../src/content/flows/rest-recovery';
import { workStressFlow } from '../src/content/flows/work-stress';
import { resourcesContent } from '../src/content/resources/resources';
import { parseGuidedFlow } from '../src/domain/flow-engine/parseFlow';
import { validateFlow } from '../src/domain/flow-engine/validateFlow';
import type { ChoiceFlowNode, FlowEffect, FlowNode, GuidedFlow, ResultFlowNode } from '../src/domain/flow-engine/types';

function isMainModule() {
  return process.argv[1] ? fileURLToPath(import.meta.url) === path.resolve(process.argv[1]) : false;
}

function getRepoRoot() {
  return path.resolve(path.dirname(fileURLToPath(import.meta.url)), '..');
}

export function loadRegisteredFlows(rootDir = getRepoRoot()) {
  const flowsDirectory = path.join(rootDir, 'src', 'content', 'flows');
  const jsonFlows = existsSync(flowsDirectory)
    ? readdirSync(flowsDirectory)
        .filter((fileName) => fileName.endsWith('.json'))
        .sort((left, right) => left.localeCompare(right))
        .map((fileName) => {
          const rawFlow = JSON.parse(readFileSync(path.join(flowsDirectory, fileName), 'utf8'));
          return parseGuidedFlow(rawFlow);
        })
    : [];

  return [workStressFlow, restRecoveryFlow, ...jsonFlows];
}

function hasSafetyInterruptToSupport(effects: FlowEffect[] | undefined) {
  return effects?.some(
    (effect) =>
      effect.kind === 'safety_interrupt' &&
      effect.destination === '/apoio' &&
      effect.blockResume === true &&
      effect.message.trim().length > 0,
  );
}

function getChoiceNode(flow: GuidedFlow, nodeId: string): ChoiceFlowNode | undefined {
  const node = flow.nodes[nodeId];
  return node?.kind === 'choice' ? node : undefined;
}

function getResultNodes(flow: GuidedFlow): Array<[string, ResultFlowNode]> {
  return Object.entries(flow.nodes).filter((entry): entry is [string, ResultFlowNode] => entry[1].kind === 'result');
}

function hasScoreEffect(node: FlowNode, scoreKey: string) {
  if (node.kind !== 'choice') {
    return false;
  }

  return node.options.some((option) =>
    option.effects?.some((effect) => effect.kind === 'score' && effect.scoreKey === scoreKey && effect.value === 1),
  );
}

export function validateRegisteredFlows(flows: GuidedFlow[]) {
  const errors: string[] = [];
  const flowIds = new Set<string>();
  const enteringPhraseOwners = new Map<string, string>();

  flows.forEach((flow) => {
    const validation = validateFlow(flow);
    errors.push(...validation.errors);

    if (flowIds.has(flow.id)) {
      errors.push(`Duplicate flow id "${flow.id}".`);
    }
    flowIds.add(flow.id);

    flow.entry.enteringPhrases.forEach((phrase) => {
      const normalizedPhrase = phrase.trim();
      const owner = enteringPhraseOwners.get(normalizedPhrase);

      if (owner) {
        errors.push(`Duplicate entering phrase "${normalizedPhrase}" used by flows ${owner} and ${flow.id}.`);
        return;
      }

      enteringPhraseOwners.set(normalizedPhrase, flow.id);
    });
  });

  return errors;
}

export function validateSrq20Contract(flow: GuidedFlow | undefined) {
  const errors: string[] = [];

  if (!flow) {
    return ['SRQ-20 flow is not registered.'];
  }

  if (flow.id !== 'srq20') {
    errors.push(`SRQ-20 flow must have id "srq20"; received "${flow.id}".`);
  }

  if (flow.entry.nodeId !== 'consent') {
    errors.push('SRQ-20 entry node must be consent.');
  }

  for (let questionNumber = 1; questionNumber <= 20; questionNumber += 1) {
    const nodeId = `q${questionNumber}`;
    const node = getChoiceNode(flow, nodeId);

    if (!node) {
      errors.push(`SRQ-20 must include question node ${nodeId}.`);
      continue;
    }

    if (!node.options.some((option) => option.id === 'yes')) {
      errors.push(`SRQ-20 ${nodeId} must include a yes option.`);
    }

    if (!node.options.some((option) => option.id === 'no')) {
      errors.push(`SRQ-20 ${nodeId} must include a no option.`);
    }

    if (questionNumber !== 17 && !hasScoreEffect(node, 'srq20')) {
      errors.push(`SRQ-20 ${nodeId} must include a score effect for scoreKey srq20.`);
    }
  }

  const q17Yes = getChoiceNode(flow, 'q17')?.options.find((option) => option.id === 'yes');
  if (!hasSafetyInterruptToSupport(q17Yes?.effects)) {
    errors.push('SRQ-20 q17 yes option must include a safety_interrupt effect to /apoio with blockResume enabled.');
  }

  const scoreBranch = flow.nodes['srq20-score'];
  if (scoreBranch?.kind !== 'score_branch' || scoreBranch.scoreKey !== 'srq20') {
    errors.push('SRQ-20 must include srq20-score score_branch with scoreKey srq20.');
  } else {
    const hasLowBranch = scoreBranch.branches.some((branch) => branch.min === 0 && branch.max === 6);
    const hasPossibleDistressBranch = scoreBranch.branches.some((branch) => branch.min === 7 && branch.max === 20);

    if (!hasLowBranch) {
      errors.push('SRQ-20 score branch must include a 0-6 branch.');
    }

    if (!hasPossibleDistressBranch) {
      errors.push('SRQ-20 score branch must include a 7-20 branch.');
    }
  }

  return errors;
}

export function validateResourceRecommendations(flows: GuidedFlow[], resourceIds: string[]) {
  const errors: string[] = [];
  const existingResourceIds = new Set(resourceIds);

  flows.forEach((flow) => {
    getResultNodes(flow).forEach(([nodeId, node]) => {
      node.recommendations?.forEach((resourceId) => {
        if (!existingResourceIds.has(resourceId)) {
          errors.push(`Flow ${flow.id} result node ${nodeId} recommends missing resource ${resourceId}.`);
        }
      });
    });
  });

  return errors;
}

export function validateContent() {
  const flows = loadRegisteredFlows();
  const resourceIds = resourcesContent.resources.map((resource) => resource.id);
  const srq20Flow = flows.find((flow) => flow.id === 'srq20');

  return [
    ...validateRegisteredFlows(flows),
    ...validateSrq20Contract(srq20Flow),
    ...validateResourceRecommendations(flows, resourceIds),
  ];
}

if (isMainModule()) {
  const errors = validateContent();

  if (errors.length > 0) {
    console.error(`Flow/content validation failed with ${errors.length} error(s):`);
    errors.forEach((error) => console.error(`- ${error}`));
    process.exit(1);
  }

  console.log(`Flow/content validation passed for ${loadRegisteredFlows().length} flow(s).`);
}
```
